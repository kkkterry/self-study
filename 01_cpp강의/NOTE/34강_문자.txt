// 문자
-> 숫자문자와 숫자는 다르다.
-> 문자의 종료는 0이다.
-> 저장하는 방식은 정수이지만 출력되는 것은 문자
-> char(1), wchar(2)

char c =1;		// 메모리상 기록 SOH(아세키 코드)
c = '1';			// 메모리상 기록 49(아세키 코드), '' => 문자

wc = 459;			// 메모리 안에 459에 해당되는 이진수를 넣어줌			
"459";			// ,"" => 문자열 4, 5, 9에 해당되는 아세키 코드가 
				각각의 바이트에 저장되어 있고 문자열로 보면 이어서 본다.
